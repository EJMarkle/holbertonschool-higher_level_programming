>>>Append integers to the list and print sorted
my_list = MyList()
my_list.append(1)
my_list.append(4)
my_list.append(2)
my_list.append(3)
my_list.append(5)
print(my_list)  # Expected: [1, 4, 2, 3, 5]
my_list.print_sorted()  # Expected: [1, 2, 3, 4, 5]
print(my_list)  # Expected: [1, 4, 2, 3, 5]

>>> Append integers in reverse order and print sorted
my_list = MyList()
my_list.append(5)
my_list.append(3)
my_list.append(2)
my_list.append(4)
my_list.append(1)
print(my_list)  # Expected: [5, 3, 2, 4, 1]
my_list.print_sorted()  # Expected: [1, 2, 3, 4, 5]
print(my_list)  # Expected: [5, 3, 2, 4, 1]

>>> Append a single integer and print sorted

>>> Append no integers and print sorted

>>> Append negative integers and print sorted
